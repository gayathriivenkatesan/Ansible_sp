#############when #############:
executing for different os based on when

---
- hosts: all or ubunto (mention group ,host etc..)
  become: yes (run as sudo user)
  gather_facts: yes
  vars:
    app_name: apache2
    service_name: httpd


  tasks:
    - name: Install apache
      apt:(2space nxt line)
	name: "{{ app_name }}" -->use jinja template
	state: present
      when: ansible_facts["ansible_os_family"] == "Debian" #or give only os_family

   - name: Install apache
      yum:(2space nxt line)
	name: "{{ service_name }}" -->use jinja template
	state: present
      when: ansible_facts["ansible_os_family"] == "RedHat" #or give only os_family

	path: '/tmp/{{ item }}' -->use jinja template
	state: touch
      loop:
        - file1.txt
	- file2.txt
	- file1.txt
